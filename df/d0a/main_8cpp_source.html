535 void setBinom(std::vector<int>& cft)
536 {
537  cft[0] = 1;
538 
539  for (int i = 1; i < order; ++i)
540  {
541  cft[i * order + 0] = 1;
542  cft[i * order + i] = 1;
543  for (int j = 1; j < i; ++j)
544  cft[i * order + j] = cft[(i - 1) * order + j] + cft[(i - 1) * order + (j - 1)];
545  }
546 }
